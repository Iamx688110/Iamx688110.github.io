<!-- 自动处理评论中的代码高亮和表情包 -->
<script>
// 监听 Cusdis 加载完成
document.addEventListener('cusdis:loaded', function() {
  // 代码高亮
  Prism.highlightAll();
  
  // 表情包解析
  const emojiParser = new EmojiJS();
  emojiParser.img_sets.emojione.path = 'https://cdn.jsdelivr.net/npm/emojione/assets/png/';
  emojiParser.img_sets.emojione.sheet = 64;
  
  // 替换评论中的表情符号
  setInterval(() => {
    const commentContents = document.querySelectorAll('#cusdis_thread .comment-content');
    commentContents.forEach(el => {
      if (!el.dataset.emojiParsed) {
        el.innerHTML = emojiParser.replace_colons(el.innerHTML);
        el.dataset.emojiParsed = 'true';
      }
    });
  }, 1000);
});
</script>
