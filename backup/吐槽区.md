  <!-- 功能脚本 -->
  <script src="https://cdn.jsdelivr.net/npm/prismjs@1/prism.min.js"></script>
  <script src="https://cdn.jsdelivr.net/npm/emoji-picker-element@1/dist/emoji-picker-element.min.js"></script>
  <script>
    // 1. 加载GitHub Issues评论
    function loadComments() {
      const container = document.getElementById('gmeek-comments');
      // 调用Gmeek的评论加载接口
      fetch(`https://api.github.com/repos/lamx688110/lamx688110.github.io/issues?labels=评论`)
        .then(res => res.json())
        .then(issues => {
          issues.forEach(issue => {
            const commentEl = document.createElement('div');
            commentEl.style = "padding: 15px; border-bottom: 1px solid #eee;";
            // 解析评论中的代码和表情
            let content = issue.body
              .replace(/```([\s\S]+?)```/g, '<pre><code class="language-$1">$1</code></pre>') // 代码块
              .replace(/:([a-z0-9_]+):/g, '<img src="https://cdn.jsdelivr.net/npm/emojione/assets/png/$1.png" width="20" height="20">'); // 表情包
            commentEl.innerHTML = `
              <h4>${issue.user.login}</h4>
              <div>${content}</div>
              <p style="font-size: 12px; color: #999;">${new Date(issue.created_at).toLocaleString()}</p>
            `;
            container.appendChild(commentEl);
          });
          // 代码高亮
          Prism.highlightAll();
        });
    }

    // 2. 初始化评论区
    window.onload = loadComments;
  </script>
