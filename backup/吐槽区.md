  <script>
    // 监听Cusdis加载完成
    document.addEventListener('cusdis:loaded', function() {
      // 代码高亮
      Prism.highlightAll();

      // 表情包解析
      const emojiParser = new EmojiJS();
      emojiParser.img_sets.emojione.path = 'https://cdn.jsdelivr.net/npm/emojione/assets/png/';
      emojiParser.img_sets.emojione.sheet = 64;

      // 监听评论新增并解析表情包
      const observer = new MutationObserver(mutations => {
        mutations.forEach(mutation => {
          if (mutation.addedNodes.length) {
            document.querySelectorAll('#cusdis_thread .comment-content').forEach(el => {
              if (!el.dataset.emojiParsed) {
                el.innerHTML = emojiParser.replace_colons(el.innerHTML);
                el.dataset.emojiParsed = 'true';
              }
            });
          }
        });
      });
      observer.observe(document.getElementById('cusdis_thread'), { childList: true, subtree: true });
    });
  </script>
